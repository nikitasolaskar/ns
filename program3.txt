using System;
					
public class Program
{
	public static void Main()
	{
		int i,j,k,n=5;
 for( i=0;i<n;i++)  
 {  
   for( j=0;j<i;j++)  
   {  
       Console.Write(" ");  
   }  
   for(k=1;k<=n-i;k++)  
   {  
   Console.Write("*");  
   }  
	 Console.WriteLine();
 }  
for(i=1;i<n;i++)  
{  
  for(j=1;j<n-i;j++)  
  {  
   Console.Write(" ");  
  }  
  for(k=1;k<=i+1;k++)  
  {  
   Console.Write("*");  
  }   
 Console.WriteLine();
			 }
		    Console.ReadLine();
	}
}
